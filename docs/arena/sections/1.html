def intersect_rays_1d(
    rays: Float[Tensor, "nrays 2 3"], segments: Float[Tensor, "nsegments 2 3"]
) -> Bool[Tensor, " nrays"]:
    """
    For each ray, return True if it intersects any segment.
    """
    nrays = rays.shape[0]
    nsegments = segments.shape[0]

    rays = rays[..., :2]  # nrays 2(o,d) 2 (x,y)
    segments = segments[..., :2] # nsegments 2 2

    Os, Ds = rays.transpose(0, 1)  # nrays, 2 |   nrays, 2
    L_1s, L_2s = segments.transpose(0, 1)  # nsegments, 2 | nsegments, 2
    
    # expected solution is to use einops.repeat(x, "a -> a b", b = 67)
    # t.broadcast_tensors returns views It does not usually copy data; it uses expanded views, so be careful with in-place writes.
    LHS = t.stack(t.broadcast_tensors(Ds.unsqueeze(1), (L_1s - L_2s).unsqueeze(0)), dim=-1) # nrays x n_segments x 2 x 2    
    RHS = (L_1s.unsqueeze(0) - Os.unsqueeze(1)) # nrays x nsegments x 2

    assert LHS.shape == (nrays, nsegments, 2, 2)
    assert RHS.shape == (nrays, nsegments, 2)


    singular_mask = t.linalg.det(LHS).abs() < 1e-8
    LHS[singular_mask] = t.eye(2, 2) # can it infer 2, 2 instead of me having to specify

    sol = t.linalg.solve(LHS, RHS)
    assert sol.shape == (nrays, nsegments, 2)
    print((sol[...,:1] >= 0).shape)
    print(((0 <= sol[...,1:]) & (sol[...,1:] <= 1)).shape)
    

    intersects = (sol[...,:1] >= 0) &((0 <= sol[...,1:]) & (sol[...,1:] <= 1)) # pairwise
    intersects[singular_mask] = False # remove bad ones
    intersects = intersects.any(dim=1).squeeze() 

    return intersects
